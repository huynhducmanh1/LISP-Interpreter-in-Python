(DEFINE FIZZBUZZ (X) 
    (WHILE (< X 101) 
        (BEGIN 
            (IF (= (% X 15) 0)
                (PRINT FizzBuzz)
                (IF (= (% X 3) 0)
                    (PRINT Fizz)
                    (IF (= (% X 5) 0)
                        (PRINT Buzz)
                        (PRINT X)
                    )
                )
            )
            (SET X (+ 1 X)) 
        ) 
    ) 
)
